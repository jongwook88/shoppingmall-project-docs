CREATE TABLE `CART` (
	`user_no`  int default null COMMENT '회원번호', -- 회원번호
	`product_no` int NOT NULL COMMENT '상품번호', -- 상품번호
	`nonuser_no` int default null COMMENT '비회원번호', -- 비회원번호

)
COMMENT '장바구니';

-- 장바구니
ALTER TABLE `CART`
	ADD CONSTRAINT `PK_TABLE` -- 장바구니 기본키

			`user_no`,  -- 회원번호
			`product_no`, -- 상품번호
			`nonuser_no`  -- 비회원번호
		);
